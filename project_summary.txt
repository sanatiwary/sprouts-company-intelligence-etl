# Project Objective
* Make an ETL that brings in company data (name. valuation, funding, org chart)
* See Apollo - what details they show for companies - talent intelligence, which colleges
* Work on an independent repo 
* Trigger: 300K

# Apollo Fields

| Category | Apollo Fields | Apollo Fields NOT in Current Schema | Current Schema Fields NOT IN Apollo |
| -------- | ------------- | ----------------------------------- | ----------------------------------- |
| Company Identity | Company Name, Description, Headquarters Location, Founding Year, Links | Phone Number | Crunchbase URL, GitHub Org, Domains, Citations |
| Industry & Classification | Industries, Keywords, SIC Codes, NAICS Codes | Keywords, SIC Codes, NAICS Codes | Business Model, Sub-Industry, Weighted Confidence Scores |
| Company Size | Employee Count | — | Headcount Range, Min Employees, Max Employees |
| Financial Intelligence | Funding Rounds, Funding Dates, Funding Round Type (Seed, Series A, etc.), Total Raised, Funding Stage, Number Funding Rounds | — | Revenue, Revenue Growth %, Investors, Valuation, Funding Stage History |
| Leadership & People | Employee Directory, Employee Name, Job Title, Department, Location, Seniority Status | Explicit Executive Team Model, Org Chart | Founder, Structured Org Hierarchy |
| Contact Intelligence | Email Access, Department, Seniority, Role | Contact-Level Enrichment Metadata | —
| Workforce Analytics | 6-Month Growth %, 1-Year Growth %, 2-Year Growth %, Retention, Department Trends, Location Trends | Retention Metrics | Headcount History, Employee Count by Country, Seniority Breakdown |
| Hiring Intelligence | Job Postings | — | Open Roles, Required Skills, Hiring Flow, Interview Process Structure |
| Technology Intelligence | Technology List, Technology Category, Technology Count | Technology Categories (explicit classification) | Structured Frontend/Backend/Infrastructure/SaaS buckets |
| Market Intelligence | Similar Companies | — | Product Category, Competitive Alternatives, ICP Signals, Competitive Narrative, Sourcing Insight |
| News & Activity | News | — | Funding News, Layoffs, Product Launches, Executive Changes, News Timeline |
| Social / Web Presence | Links (website, LinkedIn, Facebook, X) | Structured Social Profiles | Profile Collection |
| Open Source Intelligence | — | — | GitHub Repositories, Languages, Contributors, OSS Activity, Star Counts |
| Culture & Employer Brand | — | — | Glassdoor Rating, CEO Approval, Recommendation %, Review Themes, Interview Difficulty |
| Data Quality / Enrichment Metadata | Score, Personas, Relevant Jobs, Relevant Technology | Sales/Lead Scoring Objects | Completeness Scores, Confidence Scores, Sources Used, Enrichment Version, Raw Data |

const PDLJS = require('peopledatalabs');
require('dotenv').config();

const client = new PDLJS({
    apiKey: process.env.PDL_API_KEY
});

async function getCompany(companyName) {
    const response = await client.company.enrichment({
        name: companyName
    });

    return response;
}

module.exports = {
    getCompany
};

const fs = require('fs');
const path = require('path');
const readline = require('readline');

const extract = require('./extract');
const transform = require('./transform');
const load = require('./load');

async function run(companyName, forceRefresh) {
    try {
        const outputPath = path.join(
            __dirname,
            '..',
            'outputs',
            `${companyName}.json`
        );

        const rawData = await extract(
            companyName,
            forceRefresh
        );

        if (
            !forceRefresh &&
            fs.existsSync(outputPath) &&
            rawData.company_identity
        ) {
            console.log(
                JSON.stringify(rawData, null, 4)
            );

            console.log(
                '\nLoaded existing JSON from outputs.'
            );

            return;
        }

        const companyProfile =
            await transform(rawData);

        await load(companyProfile);

        console.log(
            JSON.stringify(
                companyProfile,
                null,
                4
            )
        );

        console.log(
            '\nCompany profile generated successfully.'
        );

    } catch (error) {
        console.error(
            'Pipeline failed:',
            error
        );
    }
}

const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
});

rl.question(
    'Enter company name (add !refresh to force API): ',
    async (input) => {

        const forceRefresh =
            input.includes('!refresh');

        const companyName =
            input.replace('!refresh', '').trim();

        await run(
            companyName,
            forceRefresh
        );

        rl.close();
    }
);

const fs = require('fs');
const path = require('path');

async function load(profile) {
    const companyName =
        profile.company_identity.company_name;

    const outputDir = './outputs';

    if (!fs.existsSync(outputDir)) {
        fs.mkdirSync(outputDir);
    }

    fs.writeFileSync(
        path.join(outputDir, `${companyName}.json`),
        JSON.stringify(profile, null, 4)
    );
}

module.exports = load;

--- FILE: ./src/extract.js ---

const fs = require('fs');
const path = require('path');

const { getCompany } = require('./providers/pdlProvider');

async function extract(companyName, forceRefresh = false) {
    const outputPath = path.join(
        __dirname,
        '..',
        'outputs',
        `${companyName}.json`
    );

    if (!forceRefresh && fs.existsSync(outputPath)) {
        console.log('\nUsing cached company profile...\n');

        return JSON.parse(
            fs.readFileSync(outputPath, 'utf8')
        );
    }

    console.log('\nFetching fresh data from PDL...\n');

    return await getCompany(companyName);
}

module.exports = extract;

--- FILE: ./src/transform.js ---

async function transform(rawData) {
    return {
        company_identity: {
            company_name:
                rawData.display_name ||
                rawData.name ||
                null,

            description:
                rawData.summary ||
                null,

            website:
                rawData.website
                    ? `https://${rawData.website}`
                    : null,

            headquarters: {
                city:
                    rawData.location?.locality ||
                    null,

                state:
                    rawData.location?.region ||
                    null,

                country:
                    rawData.location?.country ||
                    null
            },

            founded_year:
                rawData.founded ||
                null
        },

        industry: {
            industries: [
                rawData.industry,
                rawData.industry_v2
            ].filter(Boolean),

            keywords:
                rawData.tags || []
        },

        company_size: {
            employee_count:
                rawData.employee_count || null,

            employee_range:
                rawData.size || null
        },

        funding: {
            funding_stage:
                rawData.latest_funding_stage || null,

            total_funding:
                rawData.total_funding_raised || null,

            number_funding_rounds:
                rawData.number_funding_rounds || null,

            last_funding_date:
                rawData.last_funding_date || null
        }
    };
}

module.exports = transform;
